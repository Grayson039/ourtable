How to use in a prototype
<!-- 1. Include the design system --> <script src="our-table-design-system.js"></script> <!-- 2. Define your screens as functions --> <script> var screens = { 'welcome': function(theme, state) { var c = OT.c; return '<div style="height:100%;background:' + theme.bg + '">' + c.notch() + c.statusBar(theme) + c.btn(theme, 'Get Started', "otGo('home')", 'primary') + '</div>'; }, 'home': function(theme, state) { /* ... */ } }; // 3. Wire navigation (this also renders the first screen) OT.nav(screens, 'welcome', 'phone-screen', { mainScreens: ['home', 'search', 'fridge', 'list', 'profile'], state: { userName: 'Will' } }); </script> <!-- 4. The phone frame --> <div id="phone-screen" style="width:100%;height:100%;position:relative"></div>
