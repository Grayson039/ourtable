5
Wire up basic navigation
Create a stub App.tsx with the bottom tab navigator and placeholder screens — get the nav shell running before building any real screens.
After theme 
// App.tsx — minimal working nav shell
import { NavigationContainer } from '@react-navigation/native';
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import { View, Text } from 'react-native';
import { colors } from './src/theme';

const Tab = createBottomTabNavigator();

const Stub = ({ name }: { name: string }) => (
  <View style={{ flex:1, alignItems:'center', justifyContent:'center', background:colors.bg }}>
    <Text style={{ color:colors.text }}>{name}</Text>
  </View>
);

export default function App() {
  return (
    <NavigationContainer>
      <Tab.Navigator screenOptions={{ tabBarActiveTintColor: colors.primary }}>
        <Tab.Screen name="Home" component={() => <Stub name="Home" />} />
        <Tab.Screen name="Search" component={() => <Stub name="Search" />} />
        <Tab.Screen name="Fridge" component={() => <Stub name="Fridge" />} />
        <Tab.Screen name="List" component={() => <Stub name="List" />} />
        <Tab.Screen name="Me" component={() => <Stub name="Me" />} />
      </Tab.Navigator>
    </NavigationContainer>
  );
}
✅ Goal: see 5 tabs at the bottom of your phone with placeholder screens. Once this works, the scaffold is complete.

Troubleshooting
Common Issues
ErrorFix
babel-preset-expo not found Run npm install babel-preset-expo --save-dev then npx expo start --clear
Metro bundler error / cache issue npx expo start --clear — clears the Metro cache
TypeScript errors on navigation types Normal at first — add // @ts-ignore above the line and come back to type it properly later
Expo Go says "incompatible SDK" Update Expo Go on your phone from the App Store — it needs to match the SDK version
Random dependency errors after npm install You're probably in OneDrive. Move the folder out of OneDrive and run npm install again
Port already in use npx expo start --port 8082
